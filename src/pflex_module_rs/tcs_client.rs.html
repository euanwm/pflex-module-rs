15impl TCSClient {

26    /// Attempts to connect to the specified robot
27    /// # Arguments
28    /// * `ip` - IP address of the robot
29    /// * `timeout` - Optional timeout setting for all socket read/write attempts
30    pub fn connect(&mut self, ip: &str, timeout: Option<f64>) -> Result<(), std::io::Error> {
31        let timeout = timeout.unwrap_or(TCSClient::DEFAULT_TIMEOUT);
32        let addr = format!("{}:{}", ip, Self::TCS_SERVER_PORT);
33        info!("tcs_client::connect called");
34        let conn_attempt = TcpStream::connect(addr.clone());
35        match conn_attempt {
36            Ok(stream) => {
37                stream.set_read_timeout(Some(Duration::from_secs_f64(timeout)))?;
38                stream.set_write_timeout(Some(Duration::from_secs_f64(timeout)))?;
39                self.socket = Some(stream);
40                debug!("connected to client");
41                Ok(())
42            }
43            Err(e) => {
44                debug!("failed to connect to client");
45                Err(e)
46            }
47        }
48    }

50    /// Generates and sends the command payload to the robot
51    /// # Arguments
52    /// * `command` - Selected command to run from the TCSCommand enum
53    /// * `command_args` - Optional command arguments
54    /// * `wait_for_response` - A boolean option should any commands not require waiting for a response
55    /// * `read_timeout` - Optional argument to set the read timeout on the socket
56    pub fn send_command<'a>(
57        &mut self,
58        command: TCSCommand,
59        command_args: Option<Vec<&str>>,
60        wait_for_response: bool,
61        read_timeout: Option<f64>,
62    ) -> Result<Vec<String>, RobotError> {
63        info!("tcs_client::send_command called");
64        if read_timeout.is_some() {
65            self.socket
66                .as_ref()
67                .unwrap()
68                .set_read_timeout(Some(Duration::from_secs_f64(read_timeout.unwrap())))
69                .expect("Failed to set read timeout");
70        }
71        let payload: String;
72
73        // build the additional arguments if they exist
74        if command_args.is_some() {
75            let payload_slice = command_args.unwrap().join(TCSClient::SPACEBAR_SEPERATOR);
76            payload = format!(
77                "{}{}{}{}",
78                command,
79                TCSClient::SPACEBAR_SEPERATOR,
80                payload_slice,
81                TCSClient::REQUEST_SEPARATOR
82            );
83        // build the command without additional arguments
84        } else {
85            payload = format!("{}{}", command, TCSClient::REQUEST_SEPARATOR);
86        }
87        debug!("tcs_client::send_command payload: {}", payload);
88        // send the command
89        self.socket
90            .as_ref()
91            .unwrap()
92            .write_all(payload.as_bytes())
93            .expect("Failed to write message");
94
95        // read the response (if needed)
96        if wait_for_response {
97            let response = self.get_response();
98            match response {
99                Ok(r) => {
100                    let if_error_code = ResponseCodes::check_code(r[0].to_owned());
101                    match if_error_code {
102                        Ok(_) => Ok(r[1..].to_vec()),
103                        Err(e) => Err(e),
104                    }
105                }
106                Err(e) => Err(e.to_string()),
107            }
108        } else {
109            Ok(vec![])
110        }
111    }

113    fn get_response<'a>(&mut self) -> Result<Vec<String>, std::io::Error> {
114        info!("tcs_client::get_response called");
115        let read_buffer = &mut [0; 1024];
116        let mut response: Vec<u8> = Vec::new();
117        // todo: remove unwrap
118        let read_attempt = self.socket.as_ref().unwrap().read(read_buffer);
119        match read_attempt {
120            Ok(bytes_read) => {
121                response.extend_from_slice(&read_buffer[..bytes_read]);
122            }
123            Err(e) => {
124                return Err(e);
125            }
126        }
127        debug!("tcs_client::get_response payload: {:#?}", read_attempt);
128        // todo: make this nicer
129        let response_str = std::str::from_utf8(&response).unwrap();
130        let response_parts = response_str
131            .split(TCSClient::SPACEBAR_SEPERATOR)
132            .collect::<Vec<&str>>();
133        let response_parts = response_parts
134            .iter()
135            .map(|part| part.trim_end_matches(TCSClient::RESPONSE_SEPARATOR))
136            .collect::<Vec<&str>>();
137        let collated_response = response_parts
138            .iter()
139            .map(|part| part.to_string())
140            .collect::<Vec<String>>();
141        debug!(
142            "tcs_client::get_response return value: {:#?}",
143            collated_response
144        );
145        Ok(collated_response)
146    }

148    /// Closes the socket only
149    pub fn disconnect(&mut self) -> Result<(), std::io::Error> {
150        // this only closes the socket, it doesn't tell the robot that you're disconnecting
151        // that'll need to be done by calling the exit command
152        info!("tcs_client::disconnect called");
153        if self.socket.is_some() {
154            self.socket
155                .as_ref()
156                .unwrap()
157                .shutdown(Shutdown::Both)
158                .expect("Shutdown failed so I'm failing I guess(?)");
159            self.socket = None;
160            Ok(())
161        } else {
162            Err(std::io::Error::new(
163                std::io::ErrorKind::NotConnected,
164                "Not connected to a TCS",
165            ))
166        }
167    }
